import axios from 'axios';
 
export const addGame = game => (
  (dispatch) => {
    axios.post('/games/addGame', game)
      .then((res) => {
        dispatch({type: 'ADD_GAME', payload: res.data});
      }).catch((err) => {
        throw err;
      });
  }
);
