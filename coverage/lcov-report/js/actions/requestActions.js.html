import axios from 'axios';
 
export const addRequest = request => (
  (dispatch) => {
    axios.post('/requests/addRequest', request)
      .then((res) => {
        dispatch({type: 'ADD_REQUEST', payload: res.data});
      }).catch((err) => {
        throw err;
      });
  }
);
